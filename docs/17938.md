```
const findDoubleFrequency = array => {
  const doubles = new Set();
  doubles.add(0);
  let acc = 0;
  while (
    !array.find(x => {
      acc += x;
      if (doubles.has(acc)) {
        return true;
      }
      doubles.add(acc);
      return false;
    })
  ) {}
  return acc;
}; 
```
